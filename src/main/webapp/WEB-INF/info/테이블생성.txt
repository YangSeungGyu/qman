 CREATE TABLE "Q_BOARD" 
   (	
   "NO" VARCHAR2(20 BYTE) NOT NULL ENABLE, 
	"TITLE" VARCHAR2(60 BYTE) NOT NULL ENABLE, 
	"CONTENTS" CLOB, 
	"NAME" VARCHAR2(20 BYTE) NOT NULL ENABLE, 
	"ST_DT" DATE NOT NULL ENABLE, 
	"UP_DT" DATE
  )
  CREATE INDEX Q_BOARD_IDX ON Q_BOARD (NO);

   
   CREATE SEQUENCE BOARD_SQ
    START WITH 1
    INCREMENT BY 1;
